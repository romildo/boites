\section{Using the \texttt{boite} environment}

\begin{Example}
  \begin{boite}
    A single line.
  \end{boite}
\end{Example}

\begin{Example}
  \begin{boite}[boxrule=3pt,arc=3em,colframe=blue!80,colback=red!44]
    \Huge \noindent \rule{10mm}{3em} This is one single line.
  \end{boite}
\end{Example}

\begin{Example}
  \begin{boite}[leftmargin=3cm,rightmargin=1cm,title=My title]
    This is one single line with a title.
  \end{boite}
\end{Example}

\begin{Example}
  \begin{boite}
    \Blindtext[2][4]
  \end{boite}
\end{Example}
